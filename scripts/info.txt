import sys

# Add the directory '/path/to/your/module' to the Python path
sys.path.insert(0, '/path/to/your/module')  # if you have a module or package in a specific directory, and you want to import it in your script, you can use sys.path.insert to add that directory to the search path

# Now you can import modules from that directory
import your_module

The correct syntax for importing a module and calling a function would be:
import utils  # assuming you have a module named 'utils' with a function 'print_hello_world' inside it.
utils.print_hello_world()
